\def\E{3}
\def\R{25}
\def\L{0.43}
\def\g{\R/(2*\L)}
\def\C{0.00000002}
\def\Q{(1/\R)*sqrt(\L/\C)}
\def\wo{sqrt(1/(\L*\C))}
\def\w{sqrt(\wo^2 - \g^2)}
\def\xmax{2.2e-3}

\begin{document}

    \end{axis}
