\documentclass{standalone}
\usepackage{mintikz}

\begin{document}
\begin{tikzpicture}[scale=1, declare function={
				wdth = 16;
				hgth = 1.2;
			}]
	% Pka
	\foreach \A/\DA/\B/\DB/\K/\KN/\P/\PN [count=\i] in {
	\ce{CH3COOH}/{draw, fill=white}/\ce{CH3COO-}//4.8/{A,1}/+/below,
	\ce{NH4+}//\ce{NH3}/{draw, fill=white}/9.2/{A,2}/-/above}
	{
	% Acide domain
	\fill[red!20]
	(0,\P hgth) rectangle (\K,0);
	% Basic domain
	\fill[blue!10]
	(\K,0) rectangle (wdth,\P hgth);
	}
	% Common ground
	\fill[VioletRed!20]
	(4.8,hgth) rectangle (9.2,-hgth);
	\foreach \A/\DA/\B/\DB/\K/\KN/\P/\PN [count=\i] in {
	\ce{CH3COOH}/{draw, fill=white}/\ce{CH3COO-}//4.8/{A,1}/+/below,
	\ce{NH4+}//\ce{NH3}/{draw, fill=white}/9.2/{A,2}/-/above}
	{
	% Pka line
	\draw[very thick]
	(\K,0) --++ (0,\P hgth)
	node[expand style=\DA, thick, midway, left=.1] {\A}
	node[expand style=\DB, thick, midway, right=.1] {\B};
	% go through
	\draw[very thick, dashed]
	(\K,0) --++ (0,-\P hgth);
	% Plot pka with cross
	\node[\PN] at (\K,0) {$\pk[\KN]= \num{\K}$};
	\node at (\K,0) {$\times$};
	}
	% Scale
	\draw[very thick, -{Stealth[scale=1.3]}]
	(0,0) --
	node[at end, right] {$\pH$}
	(wdth,0);
	\foreach \x in {0, 5, ..., 15}{
			\draw[thick]
			(\x,0.15) -- (\x,-0.15);
		}
	\foreach \x in {0, ..., 15}{
			\draw
			(\x,0.075) -- (\x,-0.075);
		}
\end{tikzpicture}

\end{document}
