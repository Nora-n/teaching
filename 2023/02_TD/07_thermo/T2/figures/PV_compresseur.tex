\documentclass{standalone}
\usepackage{mintikz}

\begin{document}
% PV diagram - constant V
\def\N{40} % number of plot samples
\def\xmax{4}
\def\ymax{3}
\def\gam{1.2}
\def\isotherm#1#2{{ #2/(#1) }}
\def\adiabatic#1#2{{ 4*#2/(#1)^(\gam)-2}}
\begin{tikzpicture}
	\def\Th{3}
	% \def\Tc{.85}
	% \def\A{ (\Th/\Ch)^(1/(1-\gam)) }
	% \def\B{ (\Tc/\Ch)^(1/(1-\gam)) }
	\def\xa{0}
	\def\xb{0.8*\xmax}
	\def\ya{\adiabatic{\xb}{\Th}}
	\def\yb{\adiabatic{\xb}{\Th}}
	% \def\Ch{\yb*\xb^(\gam)}
	%\def\intersect#1#2{ (#1/#2)^(1/(1-\gam) }
	\coordinate (O) at (0,0);
	\coordinate (A) at (\xa,\ya);
	\coordinate (Ab) at (A|-O);
	\coordinate (B) at (\xb,\yb);
	\coordinate (Bb) at (B|-O);
	\def\xc{0.6*\xmax}
	\coordinate (C) at (\xc,\adiabatic{\xc}{\Th});
	\coordinate (Cb) at (C|-O);
	\coordinate (D) at (\xa,\adiabatic{\xc}{\Th});
	% TICKS
	% \xtick{A|-O}{$V_1$}
	\draw[dashed] (B) -- (Bb);
	\draw[dashed] (C) -- (Cb);
	\xtick{B|-O}{$V_m$}
	\xtick{C|-O}{$V_1$}
	\ytick{A-|O}{$P_0$}
	\ytick{D-|O}{$P_1$}
	% AREA
	\fill[gray, samples=\N, domain=\xb:\xc, opacity=.3]
	(A) -- (B) --
	plot (\x,\adiabatic{\x}{\Th}) --
	(D) --
	cycle;
	\node[darkgray] (W) at (barycentric cs:A=1,B=1,C=1,D=1) {\large$W$};
	% ISOTHERM
	% \draw[domain=\xb:\xa,samples=\N, smooth,
	% 	isoT, very thick, flch, mynode={.7}{left}{IsoT}]
	% plot (\x,\isotherm{\x}{\Th})
	% ;
	% ISOP, ISOV
	\draw[very thick, flch, isoP] (A) -- (B);
	\draw[very thick, flch, isoP] (C) -- (D);
	% \draw[very thick, flch, isoV, mynode={1}{above}{IsoV}] (B) -- (C);
	% ADIABATIC
	\draw[domain=\xb:\xc,samples=\N, smooth,
		adia, very thick, flch]
	plot (\x,\adiabatic{\x}{\Th});
	% POINTS
	\fill (A) circle(0.07) node[below right] {A};
	\fill (B) circle(0.07) node[below right] {B};
	\fill (C) circle(0.07) node[above right] {C};
	\fill (D) circle(0.07) node[above right] {D};
	% AXIS
	\draw[-{Stealth[scale=1.3]},thick] (0,-0.1*\ymax) -- (0,\ymax+0.1)
	node[anchor=north east] {$P$};
	\draw[-{Stealth[scale=1.3]},thick] (-0.1*\xmax,0) -- (\xmax+0.1,0)
	node[anchor=north east] {$V$};
\end{tikzpicture}

\end{document}
