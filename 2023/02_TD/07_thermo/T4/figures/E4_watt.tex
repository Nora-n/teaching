\begin{document}
% PV diagram - constant V
\def\N{40} % number of plot samples
\def\xmax{4}
\def\ymax{3}
\def\gam{2}
\def\isotherm#1#2{{ #2/(#1) }}
\def\adiabatic#1#2{{ #2/(#1)^(\gam) }}
\begin{tikzpicture}
	\def\Th{2.5}
	\def\Tc{.8}
	\def\Ch{2.4}
	\def\A{ (\Th/\Ch)^(1/(1-\gam)) }
	\def\B{ (\Tc/\Ch)^(1/(1-\gam)) }
	%\def\intersect#1#2{ (#1/#2)^(1/(1-\gam) }
	\coordinate (O) at (0,0);
	\coordinate (A) at ({\A},{\isotherm{\A}{\Th}});
  \def\xb{0.8*\xmax}
	\coordinate (C) at ({\C},{\isotherm{\C}{\Tc}});

	% ISOTHERMS
	\draw[isoT, thick,
	domain={\A}:{\xb},samples=\N,smooth]
	plot (\x,\isotherm{\x}{\Th});

	% horiz
	\path[name path=hpath, draw]
	(B) --++ (-3,0);

	% ADIABATIC
	\draw[adia,thick,flch,
		domain={\A:\B},samples=\N]
	plot (\x,\adiabatic{\x}{\Ch});

	% POINTS
	\fill (B) circle(0.07) node[right] {B};
	% \node at (B) {$\times$};
	% \node[right] at (B) {1};
	\fill (A) circle(0.07) node[left] {A};
	% \node at (A) {$\times$};
	% \node[left] at (A) {2};
	% \fill (E) circle(0.07) node[right] {C};
	% \node at (E) {$\times$};
	% \node[left] at (E) {1'};
	% AXIS
	\draw[-{Stealth[scale=1.3]},thick] (0,-0.1*\ymax) -- (0,\ymax+0.1)
	node[anchor=north east] {$P$};
	\draw[-{Stealth[scale=1.3]},thick] (-0.1*\xmax,0) -- (\xmax+0.1,0)
	node[anchor=north east] {$V$};
	% TICKS
	\xtick{A|-O}{$V_2$}
	\xtick{B|-O}{$V_1$}
	\ytick{A-|O}{$P_2$}
	\ytick{B-|O}{$P_1$}

\end{tikzpicture}
