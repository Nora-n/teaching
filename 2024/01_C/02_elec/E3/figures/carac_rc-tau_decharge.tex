    \def\TAUdraw{1.15}
    \def\E{5}
    \begin{axis}[
        xmin=-2, xmax=11,

        xlabel=$t$, ylabel=$u_C(t)$,
        xtick=\empty, ytick=\empty,
        extra y ticks={0},
        y tick label style={yshift={(\tick==0)*-.5em},

        axis lines=center,

        },
        clip=false]
        \addplot[

            smooth, thick,

        \addplot[

        \node[below=8pt] (Tg) at (T) {};
        \node[below=8pt] (Ig) at (11,0) {};
