	\def\E{5}
	\def\L{0.043}
	\def\C{0.00000002}
	\def\xmax{1e-3}
	% \def\Q{14.66}

			ymin=-5, ymax=5.2,
			xlabel=$t$, ylabel=$u_C$,
			axis lines=center,
			xtick=\empty, ytick=\empty,
			clip=false]
		% Beginning at \E
		\addplot[thick,
			domain=-1e-4:0,
			samples=500,
			black,
			forget plot]
		{\E};
		% First plot Q = 14
		\def\Ra{100}
		\def\Qa{(1/\Ra)*sqrt(\L/\C)}
		\def\wo{sqrt(1/(\L*\C))}
		\def\w{\wo*sqrt(1-1/(4*\Qa^2))}
		\addplot[thick,
			domain=0:9e-4,
			samples=500,
			Red!70]
		{\E*exp(-(\wo*\x)/(2*\Qa))*(cos(\w*\x r)+(\wo/(2*\Qa*\w))*sin(\w*\x r))};
		\addlegendentry{$Q = \fpeval{round(\Qa,2)}$}
		% Second plot Q = 2
		\def\Rb{500}
		\def\Qb{(1/\Rb)*sqrt(\L/\C)}
		\def\wo{sqrt(1/(\L*\C))}
		\def\w{\wo*sqrt(1-1/(4*\Qb^2))}
		\addplot[thick,
			domain=0:9e-4,
			samples=500,
			ForestGreen]
		{\E*exp(-(\wo*\x)/(2*\Qb))*(cos(\w*\x r)+(\wo/(2*\Qb*\w))*sin(\w*\x r))};
		\addlegendentry{$Q = \fpeval{round(\Qb,2)}$}
		% Third plot Q = 0.5
		\def\Rc{2932}
		\def\Qc{(1/\Rc)*sqrt(\L/\C)}
		\def\wo{sqrt(1/(\L*\C))}
		\def\w{\wo*sqrt(1-1/(4*\Qc^2))}
		\addplot[thick,
			domain=0:9e-4,
			samples=500,
			brandeisblue]
		{\E*(\wo*\x+1)*exp(-\wo*\x)};
		\addlegendentry{$Q = \fpeval{round(\Qc,2)}$}
		% Fourth plot Q = 0.2
		\def\Rd{7500}
		\def\Qd{(1/\Rd)*sqrt(\L/\C)}
		\def\Qeff{\fpeval{\Qd}}
		\def\wo{sqrt(1/(\L*\C))}
		\def\rp{-(\wo/(2*\Qeff))+\wo*sqrt(1/(4*\Qeff^2)-1)}
		\def\rpeff{\fpeval{\rp}}
		\def\rl{-(\wo/(2*\Qeff))-\wo*sqrt(1/(4*\Qeff^2)-1)}
		\def\rleff{\fpeval{\rl}}
		\def\rdif{\fpeval{\rpeff-\rleff}}
		\addplot[thick,
			domain=0:9e-4,
			samples=500,
			Purple!70]
		{(\E/(\rdif))*(\rpeff*exp(\rleff*\x)-\rleff*exp(\rpeff*\x))};
		\addlegendentry{$Q = \fpeval{round(\Qd,2)}$}
		% \node[anchor=north east, align=right] (RQtext) at (axis cs: \xmax, 5) {
