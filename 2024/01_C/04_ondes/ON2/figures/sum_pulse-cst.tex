\documentclass[border=3pt,tikz]{standalone}
\usepackage{mintikz}
\tikzset{>=latex} % for LaTeX arrow head

\colorlet{xcol}{blue!70!black}
\tikzstyle{vvec}=[->,myred,very thick,line cap=round]
\tikzstyle{rope}=[brown!20!black,double=brown!70!black,double distance=1,line width=0.3] %very thick
\tikzstyle{brope}=[black,double=black,double distance=1,line width=0.3] %very thick
\def\wave#1#2{
  ({(#1-0.22)*\xmax},0) to[out=0,in=180,looseness=0.6] (#1*\xmax,#2*\ymax)
  to[out=0,in=180,looseness=0.6]++ (0.22*\xmax,#2*-\ymax)
}

\def\xmax{3.0}
\def\ymax{0.8}
\def\hsht{2}

\begin{tikzpicture}
	% SUPERPOSITION, CONSTRUCTIVE - t = 0
	\def\xL{-0.5}
	\def\xR{0.5}
	\coordinate (L) at (\xL*\xmax,\ymax);
	\coordinate (R) at (\xR*\xmax,\ymax);
	\node[above=.1, \sswitch{white}{black}] at (L) {$f(x-ct)$};
	\node[above=.1, \sswitch{white}{black}] at (R) {$f(x+ct)$};
	\draw[\sswitch{white}{brope}] %very thick,xcol]
	(-\xmax,0) -- \wave{\xL}{1} -- \wave{\xR}{1} -- (\xmax,0);
	\draw[\sswitch{white}{vvec}]
	(L)++(-65:0.5*\ymax) --++ (0.17*\xmax,0) node[right=.1] {$c$};
	\draw[\sswitch{white}{vvec}]
	(R)++(-115:0.5*\ymax) --++ (-0.17*\xmax,0) node[left=.1] {$c$};
	\node[right,scale=0.9] at (-1.35*\xmax,0.6*\ymax) {$t=0$};
	% SUPERPOSITION, CONSTRUCTIVE - t = Dt
	\begin{scope}[shift={(0,-\hsht)}]
		\coordinate (T) at (0,\ymax);
		\draw[\sswitch{white}{dashed}] \wave{0}{1};
		\draw[\sswitch{white}{brope}] %very thick,xcol]
		(-\xmax,0) -- (-0.22*\xmax,0) to[out=0,in=180,looseness=0.4] (0,2*\ymax)
		to[out=0,in=180,looseness=0.4] (0.22*\xmax,0) -- (\xmax,0);
		\draw[\sswitch{white}{vvec}]
		(T)++(-60:0.5*\ymax) --++ (0.17*\xmax,0) node[right=-2] {$c$};
		\draw[\sswitch{white}{vvec}]
		(T)++(-120:0.5*\ymax) --++ (-0.17*\xmax,0) node[left=-2] {$c$};
		\node[right,scale=0.9] at (-1.35*\xmax,0.6*\ymax) {$t=\Delta t$};
	\end{scope}
	% SUPERPOSITION, CONSTRUCTIVE - t = 2*Dt
	\begin{scope}[shift={(0,-2*\hsht)}]
		\def\xL{-0.5}
		\def\xR{0.5}
		\coordinate (L) at (\xL*\xmax,\ymax);
		\coordinate (R) at (\xR*\xmax,\ymax);
		\draw[\sswitch{white}{brope}] %very thick,xcol]
		(-\xmax,0) -- \wave{\xL}{1} -- \wave{\xR}{1} -- (\xmax,0);
		\draw[\sswitch{white}{vvec}]
		(R)++(-65:0.5*\ymax) --++ (0.17*\xmax,0) node[right=.1] {$c$};
		\draw[\sswitch{white}{vvec}]
		(L)++(-115:0.5*\ymax) --++ (-0.17*\xmax,0) node[left=.1] {$c$};
		\node[right,scale=0.9] at (-1.35*\xmax,0.6*\ymax) {$t=2\Delta t$};
	\end{scope}
\end{tikzpicture}
