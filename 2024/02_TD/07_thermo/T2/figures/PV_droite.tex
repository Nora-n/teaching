\documentclass{standalone}
\usepackage{mintikz}

\begin{document}
% PV diagram - constant V
\def\N{40} % number of plot samples
\def\xmax{4}
\def\ymax{3}
\def\gam{1.2}
\def\isotherm#1#2{{ #2/(#1) }}
\def\adiabatic#1#2{{ #2/(#1)^(\gam) }}
\begin{tikzpicture}
	\def\Th{1.8}
	% \def\Tc{.85}
	% \def\A{ (\Th/\Ch)^(1/(1-\gam)) }
	% \def\B{ (\Tc/\Ch)^(1/(1-\gam)) }
	\def\xa{0.2*\xmax}
	\def\xb{0.8*\xmax}
	\def\ya{\isotherm{\xa}{\Th}}
	\def\yb{\isotherm{\xb}{\Th}}
	\def\Ch{\yb*\xb^(\gam)}
	%\def\intersect#1#2{ (#1/#2)^(1/(1-\gam) }
	\coordinate (O) at (0,0);
	\coordinate (A) at (\xa,\ya);
	\coordinate (Ab) at (A|-O);
	\coordinate (B) at (\xb,\yb);
	\coordinate (Bb) at (B|-O);
	\coordinate (C) at (A|-B);
	% \coordinate (D) at (\xa,\adiabatic{\xa}{\Th});
	% AREA
	\fill[gray, opacity=.3]
	(C)
	rectangle (B|-0,0)
	node[midway, scale=1.3, darkgray, text opacity=1]
		{$W_{3}$};
	\fill[gray, samples=\N, domain=\xa:\xb, opacity=.3]
	plot (\x,\isotherm{\x}{\Th}) --
	(C) --
	(A);
	\node[darkgray]
	(W13) at ([shift={(-.3,-.3)}]barycentric cs:A=1,B=1,C=1)
	{$\Delta{W}_{13}$};
	\fill[gray, samples=\N, domain=\xb:\xa, opacity=.3]
	(A) --
	(B) --
	plot (\x,\isotherm{\x}{\Th});
	\node[darkgray, rotate=-35] (W12) at ([shift={(-.1,-.2)}]$(A)!0.5!(B)$) {\small$\Delta{W_{12}}$};
	% ISOTHERM
	\draw[domain=\xb:\xa,samples=\N, smooth,
		isoT, very thick, flch]
	plot (\x,\isotherm{\x}{\Th})
	;
	% ISOP, ISOV
	\draw[very thick, flch, isoP] (B) -- (C);
	\draw[very thick, flch, isoV] (C) -- node[miway, left] {A3B} (A);
	% ADIABATIC
	% \draw[domain=\xb:\xa,samples=\N, smooth,
	% 	adia, very thick, flch, mynode={.8}{right}{Adia}]
	% plot (\x,\adiabatic{\x}{\Ch});
	% LINE
	\draw[very thick, flch, orange]
	(B) --
	node[miway, above, sloped] {A2B}
	(A);
	% POINTS
	\fill (A) circle(0.07) node[above, anchor=south] {B};
	\fill (B) circle(0.07) node[right, anchor=west] {A};
	\node[isoT, fill=white, fill opacity=.7, text opacity=1, inner sep=.5]
	at ([shift={(.7,-.3)}]A)
	{A1B};
	% AXIS
	\draw[-{Stealth[scale=1.3]},thick] (0,-0.1*\ymax) -- (0,\ymax+0.1)
	node[anchor=north east] {$P$};
	\draw[-{Stealth[scale=1.3]},thick] (-0.1*\xmax,0) -- (\xmax+0.1,0)
	node[anchor=north east] {$V$};
	% TICKS
	\xtick{A|-O}{$V_1$}
	\xtick{B|-O}{$V_2$}
	\ytick{A-|O}{$P_1$}
	\ytick{B-|O}{$P_2$}
\end{tikzpicture}

\end{document}
