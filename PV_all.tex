% \def\isotherm#1#2{{ #2/(#1) }}
\begin{tikzpicture}[declare function={%
				isotherm(\V,\T) = \T/(\V)}]
	% VALUES
	\def\T{1.2}
	\def\xa{0.2*\xmax}
	\def\xb{0.8*\xmax}
	\def\ya{isotherm(\xa,\T)}
	\def\yb{isotherm(\xb,\T)}
	% AREA
	\coordinate (O) at (0,0);
	\coordinate (A) at (\xa,\ya);
	\coordinate (B) at (\xb,\yb);
	\fill[gray,samples=\N,domain=\xa:\xb]
	plot(\x,isotherm(\x,\T)) --
	(B|-0,0) --
	(A|-0,0)
	node[midway,darkgray] {$W$} --
	cycle;
	\fill (A) circle(0.07) node[above, anchor=south] {A};
	\fill (D) circle(0.07) node[below left] {D};
	\fill (B) circle(0.07) node[right, anchor=west] {B};
	% % ISOTHERM

	% plot (\x,isotherm(\x,\T));

	node[anchor=north east] {$P$};
	\draw[-{Stealth[scale=1.3]},thick] (-0.1*\xmax,0) -- (\xmax+0.1,0)
	node[anchor=north east] {$V$};
	% TICKS
	\xtick{A|-O}{$V_1$}
	\xtick{B|-O}{$V_2$}
	\ytick{A-|O}{$P_1$}
	\ytick{B-|O}{$P_2$}
